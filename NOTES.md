- -0x6C = DR
- -0x6B = DM
- -0x6A = ML
- -0x69 = MC
- -0x68 = MR
- -0x67 = AML
- -0x66 = AMC
- -0x65 = AMR
- -0x64 = ST
- -0x63 = WBL
- -0x62 = WBR
- -0x61 = attribute start?
- 0 = base
- +0x0C = Random ID
- +0x10 = UID
